async function loadFromGist() { if (!githubToken || !gistId) return; try { const response = await fetch(`https://api.github.com/gists/${gistId}`, { headers: { 'Authorization': `token ${githubToken}` } }); if (!response.ok) { console.error("Gagal mengambil data dari Gist"); return; } const gist = await response.json(); if (!gist.files['absensi-data.json']) { console.error("File absensi-data.json tidak ditemukan di Gist"); return; } const fileContent = gist.files['absensi-data.json'].content; const parsed = JSON.parse(fileContent); allRecords = parsed.data || []; renderRecords(); console.log(`✅ Loaded ${allRecords.length} records dari GitHub Gist`); } catch (error) { console.error("❌ Error loadFromGist:", error); } }
